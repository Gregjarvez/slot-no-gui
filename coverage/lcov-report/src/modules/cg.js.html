 |---%s--- | ---%s--- | ---%s--- |
`;
 
  var symbols = ['bll', 'che', 'org', 'app', 'sev'];
 
  function reels(reelsArray) {
    var each = 0;
    var display = [reelStructure];
 
    for (var i = 0; i < 9; i++) {
      var j = i % reelsArray.length;
      display.push(symbols[reelsArray[j][each] - 1]);
      if (j === 2) each++;

    return console.log.apply(null, display);
  }
